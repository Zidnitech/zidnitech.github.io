01 // Overview

This repository implements a production-grade WhatsApp automation engine designed for e-commerce operations. It handles real-time WooCommerce order events, processes them through a Redis message queue, and dispatches contextual WhatsApp messages via the Meta WhatsApp Cloud API.

The system is stateless, horizontally scalable, and built around event-driven architecture — meaning every message dispatched is triggered by a verified business event, not a cron job or manual trigger.

Why this exists: Most WhatsApp automation tutorials show you how to send a message. This guide shows you how to build the infrastructure that sends the right message at the right time to the right person — reliably, at scale, in production.

04 // Webhook Handler

The webhook handler receives events from WooCommerce and Meta (for inbound WhatsApp messages), verifies their HMAC signatures, normalizes the payload, and enqueues a job. It returns 200 immediately — all processing happens asynchronously.

JS src/webhooks/woocommerce.js 
const crypto = require('crypto');
const { Queue } = require('bullmq');
const logger = require('../utils/logger');

const messageQueue = new Queue('whatsapp-dispatch', {
  connection: { url: process.env.REDIS_URL },
  defaultJobOptions: {
    attempts: 3,
    backoff: { type: 'exponential', delay: 5000 }
  }
});

/**
 * Verify WooCommerce HMAC-SHA256 signature
 */
function verifyWooSignature(rawBody, signature) {
  const expected = crypto
    .createHmac('sha256', process.env.WC_WEBHOOK_SECRET)
    .update(rawBody)
    .digest('base64');
  return crypto.timingSafeEqual(
    Buffer.from(expected),
    Buffer.from(signature)
  );
}

/**
 * WooCommerce order.created webhook
 * Maps to: /webhooks/woocommerce/order-created
 */
async function handleOrderCreated(req, res) {
  const rawBody = req.rawBody;
  const sig = req.headers['x-wc-webhook-signature'];

  if (!verifyWooSignature(rawBody, sig)) {
    logger.warn('Invalid WooCommerce signature', { ip: req.ip });
    return res.status(401).end();
  }

  const order = req.body;
  const phone = normalizePhone(order.billing.phone);

  if (!phone) {
    logger.info('Order has no phone — skipping', { orderId: order.id });
    return res.status(200).end();
  }

  // Determine automation flow based on payment method
  const flow = order.payment_method === 'cod'
    ? 'cod_confirmation'
    : 'order_confirmation';

  await messageQueue.add(flow, {
    to: phone,
    orderId: order.id,
    orderNumber: order.number,
    total: order.total,
    currency: order.currency,
    items: order.line_items.map(i => i.name),
    customerName: order.billing.first_name
  }, {
    jobId: `order-${order.id}`,  // dedup key
    delay: 0
  });

  logger.info('Order enqueued', { orderId: order.id, flow });
  res.status(200).end();
}

// E.164 normalization — Saudi numbers example
function normalizePhone(raw) {
  if (!raw) return null;
  const digits = raw.replace(/\D/g, '');
  if (digits.startsWith('966'))  return digits;
  if (digits.startsWith('05'))   return '966' + digits.slice(1);
  if (digits.startsWith('5'))    return '966' + digits;
  return null;
}

module.exports = { handleOrderCreated };

05 // Queue Worker

The BullMQ worker processes jobs from the Redis queue. Each job type maps to a specific message flow. Workers are stateless — you can run multiple instances behind a load balancer without coordination.

JS src/workers/dispatch.worker.js 
const { Worker } = require('bullmq');
const whatsapp = require('../services/whatsapp.service');
const flows = require('../flows');
const logger = require('../utils/logger');

const worker = new Worker('whatsapp-dispatch', async (job) => {

  logger.info(`Processing job: ${job.name}`, { id: job.id });

  switch (job.name) {
    case 'order_confirmation':
      return await flows.orderConfirmation(job.data);

    case 'cod_confirmation':
      return await flows.codConfirmation(job.data);

    case 'cart_recovery_1':
      return await flows.cartRecovery(job.data, { step: 1 });

    case 'cart_recovery_2':
      return await flows.cartRecovery(job.data, { step: 2 });

    case 'review_request':
      return await flows.reviewRequest(job.data);

    case 'shipment_update':
      return await flows.shipmentUpdate(job.data);

    default:
      logger.warn(`Unknown job type: ${job.name}`);
  }

}, {
  connection: { url: process.env.REDIS_URL },
  concurrency: 10   // process 10 jobs concurrently per worker
});

worker.on('failed', (job, err) => {
  logger.error('Job failed', {
    job: job.name, attempt: job.attemptsMade,
    error: err.message
  });
});

06 // Template Manager

All outbound messages use Meta-approved templates. The template service handles parameter injection, language selection, and constructs the correct Meta API payload structure.

Meta requirement: Any message initiated by your business (outside an active 24-hour customer session) must use a pre-approved template. Templates are submitted via the Meta Business Manager and typically approved within minutes to a few hours.
JS src/services/whatsapp.service.js 
const axios = require('axios');

const BASE_URL = `https://graph.facebook.com/v19.0/${process.env.WHATSAPP_PHONE_NUMBER_ID}/messages`;

/**
 * Send a pre-approved template message
 * @param {string} to  - E.164 phone number (e.g. "966501234567")
 * @param {string} templateName - Approved template name
 * @param {string} lang  - Language code ("ar" | "en_US")
 * @param {Array}  components - Header/body/button parameter arrays
 */
async function sendTemplate(to, templateName, lang, components = []) {
  const payload = {
    messaging_product: 'whatsapp',
    to: to,
    type: 'template',
    template: {
      name: templateName,
      language: { code: lang },
      components: components
    }
  };

  const response = await axios.post(BASE_URL, payload, {
    headers: {
      'Authorization': `Bearer ${process.env.WHATSAPP_TOKEN}`,
      'Content-Type': 'application/json'
    }
  });

  return response.data;
}

/**
 * Send interactive message with reply buttons
 */
async function sendButtons(to, body, buttons) {
  const payload = {
    messaging_product: 'whatsapp',
    to: to,
    type: 'interactive',
    interactive: {
      type: 'button',
      body: { text: body },
      action: {
        buttons: buttons.map((b, i) => ({
          type: 'reply',
          reply: { id: `btn_${i}`, title: b }
        }))
      }
    }
  };
  return axios.post(BASE_URL, payload, {
    headers: { 'Authorization': `Bearer ${process.env.WHATSAPP_TOKEN}` }
  });
}

module.exports = { sendTemplate, sendButtons };

07 // Automation Flows

Each flow is a pure function that receives job data and calls the WhatsApp service. Flows with multiple steps use BullMQ's delay option to schedule follow-up messages.

Cart Recovery — 3-Step Sequence

JS src/flows/cartRecovery.js 
const { Queue } = require('bullmq');
const wa = require('../services/whatsapp.service');

const queue = new Queue('whatsapp-dispatch', {
  connection: { url: process.env.REDIS_URL }
});

const DELAYS = {
  step1: 30 * 60 * 1000,   // 30 minutes
  step2: 24 * 60 * 60 * 1000 // 24 hours
};

async function cartRecovery(data, { step }) {
  const { to, cartUrl, productName, customerName } = data;

  if (step === 1) {
    // Step 1: Soft reminder with discount
    await wa.sendTemplate(to, 'cart_recovery_v1', 'ar', [{
      type: 'body',
      parameters: [
        { type: 'text', text: customerName },
        { type: 'text', text: productName },
        { type: 'text', text: 'COMPLETE10' }
      ]
    }, {
      type: 'button',
      sub_type: 'url',
      index: 0,
      parameters: [{ type: 'text', text: cartUrl }]
    }]);

    // Schedule step 2 if no action taken
    await queue.add('cart_recovery_2', data, {
      jobId: `cart-step2-${data.cartId}`,
      delay: DELAYS.step2
    });
  }

  if (step === 2) {
    // Step 2: Scarcity angle / open question
    await wa.sendTemplate(to, 'cart_recovery_v2', 'ar', [{
      type: 'body',
      parameters: [{ type: 'text', text: customerName }]
    }]);
  }
}

module.exports = { cartRecovery };
Deduplication: The jobId on each queued job ensures that if a webhook fires twice (WooCommerce sometimes retries), only one message sequence is dispatched per cart. BullMQ will silently drop the duplicate if the jobId already exists.

08 // API Reference

The engine exposes a small internal REST API for manual triggers, health checks, and integration with external tools like Zapier.

POST /webhooks/woocommerce/order-created
Receives WooCommerce order.created webhook. Verifies HMAC-SHA256 signature, normalizes phone, routes to order_confirmation or cod_confirmation flow.

Headers: X-WC-Webhook-Signature (base64 HMAC)
POST /webhooks/cart-abandoned
Receives abandoned cart event from WooCommerce Abandoned Cart plugin or custom trigger. Enqueues cart_recovery_1 with 30-minute delay.

Body: { cartId, customerPhone, customerName, items, cartUrl }
POST /trigger/:flowName
Manual trigger endpoint — useful for Zapier integration or testing. Enqueues any supported flow by name.

Body: Flow-specific data object. Returns { jobId, queued: true }.
GET /health
Health check. Returns Redis connection status, queue depth, and worker count. Use for load balancer probes.
GET /webhooks/meta
Meta webhook verification endpoint (GET). Returns hub.challenge if hub.verify_token matches WEBHOOK_VERIFY_TOKEN.
