// 02 Setting Up the Webhook Server

Install dependencies and spin up an Express server that handles both the verification handshake and incoming message events.

bash — install deps
npm init -y

server.js — Node.js / Express
require('dotenv').config();
const express    = require('express');
const bodyParser = require('body-parser');
const { handleWebhook } = require('./handler');

const app  = express();
const PORT = process.env.PORT || 3000;
app.use(bodyParser.json());

// ── Verification handshake (Meta calls this once on setup) ───
app.get('/webhook', (req, res) => {
  const mode      = req.query['hub.mode'];
  const token     = req.query['hub.verify_token'];
  const challenge = req.query['hub.challenge'];
  if (mode === 'subscribe' && token === process.env.VERIFY_TOKEN) {
    console.log('✅ Webhook verified');
    res.status(200).send(challenge);
  } else {
    res.sendStatus(403);
  }
});

// ── Receive events ───────────────────────────────────────────
app.post('/webhook', async (req, res) => {
  res.sendStatus(200); // respond immediately, process async
  try {
    await handleWebhook(req.body);
  } catch (err) {
    console.error('Webhook error:', err.message);
  }
});

app.listen(PORT, () => console.log(`🚀 Server on port ${PORT}`));

// 03 Parsing WhatsApp Webhook Payloads

Meta's payload is deeply nested. Here's the exact extraction logic for phone number, message body, contact name, and timestamp.

json — raw Meta payload sample
{
  "object": "whatsapp_business_account",
  "entry": [{
    "id": "WHATSAPP_BUSINESS_ACCOUNT_ID",
    "changes": [{
      "value": {
        "messaging_product": "whatsapp",
        "contacts": [{ "profile": { "name": "Ahmed Al-Rashid" }, "wa_id": "966501234567" }],
        "messages": [{
          "id": "wamid.xxxx",
          "from": "966501234567",
          "timestamp": "1735689600",
          "type": "text",
          "text": { "body": "السلام عليكم، ابغى عرض أسعار" }
        }]
      }
    }]
  }]

parser.js — extract lead data cleanly
/**
 * Extracts structured lead data from a Meta webhook payload.
 * Handles: text, button, interactive (reply/list), and media messages.
 * Returns null for non-message events (status updates, etc.)
 */
function extractLead(body) {
  const value   = body?.entry?.[0]?.changes?.[0]?.value;
  const msg     = value?.messages?.[0];
  const contact = value?.contacts?.[0];
  if (!msg) return null;

  const type = msg.type;
  let msgBody = '';

  switch (type) {
    case 'text':
      msgBody = msg.text?.body; break;
    case 'button':
      msgBody = `[BTN] ${msg.button?.text}`; break;
    case 'interactive': {
      const ia = msg.interactive;
      msgBody = ia?.button_reply?.title || ia?.list_reply?.title || '[interactive]';
      break;
    }
    default:
      msgBody = `[${type}]`; // image, audio, document, etc.
  }

  return {
    messageId : msg.id,
    phone     : msg.from,                   // "966501234567"
    name      : contact?.profile?.name || 'Unknown',
    body      : msgBody || '',
    type,
    timestamp : new Date(+msg.timestamp * 1000).toISOString(),
    source    : 'whatsapp',
  };
}

module.exports = { extractLead };

// 04 Push Leads to Google Sheets

Two approaches: Google Apps Script (zero infra) or the Sheets API v4 directly from Node.js.

Option A — Google Apps Script (Free, Serverless)

Apps Script — Code.gs
// Deploy: Extensions → Apps Script → Deploy → Web App
// Execute as: Me | Access: Anyone

const SHEET_NAME = 'Leads';
const SECRET     = 'your_apps_script_secret';

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    if (data.secret !== SECRET)
      return ContentService.createTextOutput('Unauthorized');

    const sheet = SpreadsheetApp.getActiveSpreadsheet()
                                .getSheetByName(SHEET_NAME);

    if (sheet.getLastRow() === 0) {
      sheet.appendRow(['Timestamp','Phone','Name','Message','Type','Source','Stage']);
    }

    sheet.appendRow([
      data.timestamp, data.phone, data.name,
      data.body, data.type, data.source, 'new_lead'
    ]);

    return ContentService
      .createTextOutput(JSON.stringify({ ok: true }))
      .setMimeType(ContentService.MimeType.JSON);

  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({ error: err.message }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

// 05 Push Leads to HubSpot CRM

Create or update a HubSpot Contact on every lead using upsert logic — same phone never creates a duplicate.

hubspot.js
const axios = require('axios');
const BASE  = 'https://api.hubapi.com';
const HDR   = () => ({
  Authorization: `Bearer ${process.env.HUBSPOT_API_KEY}`,
  'Content-Type': 'application/json',
});

async function pushToHubspot(lead) {
  const phone = '+' + lead.phone;

  // 1. Search contact by phone ───────────────────────────────
  const { data: sr } = await axios.post(
    `${BASE}/crm/v3/objects/contacts/search`,
    { filterGroups: [{ filters: [
      { propertyName:'phone', operator:'EQ', value: phone }
    ]}], properties:['phone'] },
    { headers: HDR() }
  );

  if (sr.results?.length) {
    // 2a. UPDATE ───────────────────────────────────────────────
    await axios.patch(
      `${BASE}/crm/v3/objects/contacts/${sr.results[0].id}`,
      { properties: {
        whatsapp_last_message : lead.body,
        whatsapp_last_contact : lead.timestamp,
        hs_lead_status        : 'IN_PROGRESS',
      }},
      { headers: HDR() }
    );
    console.log('🔄 HubSpot updated:', sr.results[0].id);
  } else {
    // 2b. CREATE ───────────────────────────────────────────────
    const { data: nc } = await axios.post(
      `${BASE}/crm/v3/objects/contacts`,
      { properties: {
        firstname              : lead.name,
        phone,
        lifecyclestage         : 'lead',
        hs_lead_status         : 'NEW',
        lead_source            : 'WhatsApp',
        whatsapp_first_message : lead.body,
        whatsapp_first_contact : lead.timestamp,
      }},
      { headers: HDR() }
    );
    console.log('➕ HubSpot created:', nc.id);
  }
}

module.exports = { pushToHubspot };

// 07 Full Production Handler

Combining all modules with idempotency guard, parallel dispatch, and per-destination error isolation.

handler.js — production-ready
const { extractLead }  = require('./parser');
const { pushToSheets }  = require('./sheets');
const { pushToHubspot } = require('./hubspot');

// In-memory dedup (swap for Redis in production)
const seen = new Set();

async function handleWebhook(body) {
  const lead = extractLead(body);
  if (!lead) return; // status update or unsupported event

  // Idempotency — Meta can re-deliver the same event
  if (seen.has(lead.messageId)) {
    console.log('⚠️  Dup skipped:', lead.messageId);
    return;
  }
  seen.add(lead.messageId);
  if (seen.size > 10000) seen.clear(); // memory guard

  console.log(`📥 [${lead.timestamp}] ${lead.phone} → ${lead.body.slice(0,80)}`);

  // Fire both destinations in parallel; fail one, keep the other
  const results = await Promise.allSettled([
    pushToSheets(lead),
    pushToHubspot(lead),
  ]);

  results.forEach((r, i) => {
    if (r.status === 'rejected')
      console.error(`❌ ${i===0?'Sheets':'HubSpot'}:`, r.reason?.message);
  });
}

module.exports = { handleWebhook };

// 08 Python / Flask Alternative

pip install flask requests gspread google-auth
app.py — Flask webhook server
import os, threading
from flask import Flask, request, jsonify
import gspread
from google.oauth2.service_account import Credentials
from datetime import datetime, timezone

app = Flask(__name__)
VERIFY_TOKEN = os.environ["VERIFY_TOKEN"]
SHEETS_ID    = os.environ["GOOGLE_SHEETS_ID"]
seen: set    = set()

def get_sheet():
    creds = Credentials.from_service_account_file(
        "service-account.json",
        scopes=["https://www.googleapis.com/auth/spreadsheets"]
    )
    return gspread.authorize(creds).open_by_key(SHEETS_ID).worksheet("Leads")

def extract(body):
    try:
        v   = body["entry"][0]["changes"][0]["value"]
        msg = v["messages"][0]
        c   = v.get("contacts",[{}])[0]
    except (KeyError,IndexError): return None
    t   = msg.get("type","unknown")
    txt = msg.get("text",{}).get("body") or f"[{t}]"
    ts  = datetime.fromtimestamp(int(msg["timestamp"]),tz=timezone.utc).isoformat()
    return dict(id=msg["id"],phone=msg["from"],
                name=c.get("profile",{}).get("name","Unknown"),
                body=txt,type=t,timestamp=ts,source="whatsapp")

def process(body):
    lead = extract(body)
    if not lead or lead["id"] in seen: return
    seen.add(lead["id"])
    try:
        get_sheet().append_row([
            lead["timestamp"],lead["phone"],lead["name"],
            lead["body"],lead["type"],lead["source"],"new_lead"
        ])
        print(f"✅ {lead['phone']} → {lead['body'][:60]}")
    except Exception as e: print(f"❌ {e}")

@app.route("/webhook", methods=["GET"])
def verify():
    if (request.args.get("hub.mode") == "subscribe" and
            request.args.get("hub.verify_token") == VERIFY_TOKEN):
        return request.args.get("hub.challenge"), 200
    return "Forbidden", 403

@app.route("/webhook", methods=["POST"])
def webhook():
    threading.Thread(target=process,args=(request.get_json(force=True),),daemon=True).start()
    return "", 200

if __name__ == "__main__": app.run(port=3000)

// 09 Security & Signature Verification

Production webhooks need HMAC signature verification to reject spoofed requests.

security.js — verify Meta signature
const crypto = require('crypto');

/**
 * Meta signs every POST with X-Hub-Signature-256.
 * Verify before processing — prevents spoofed/forged requests.
 * Usage: app.use(bodyParser.json({ verify: verifySignature }))
 */
function verifySignature(req, res, buf) {
  const sig = req.headers['x-hub-signature-256'];
  if (!sig) return; // skip for local dev (ngrok)

  const expected = 'sha256=' + crypto
    .createHmac('sha256', process.env.WHATSAPP_APP_SECRET)
    .update(buf)
    .digest('hex');

  if (!crypto.timingSafeEqual(Buffer.from(sig), Buffer.from(expected))) {
    throw new Error('Invalid signature');
  }
}

module.exports = { verifySignature };
