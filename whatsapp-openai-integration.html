Working code — Node.js and Python — for building an AI-powered WhatsApp chatbot with conversation memory, human handover, and interactive buttons.

Customer sends message → Webhook receiver → Load conversation history → OpenAI API → WhatsApp reply → Save history

00 Prerequisites

  • Node.js 18+ or Python 3.10+
  • WhatsApp Business API access (via any BSP or Meta Cloud API)
  • OpenAI API key — platform.openai.com
  • Public HTTPS endpoint (use ngrok for local development)
  • Redis (optional — for production conversation storage)

// Main server — index.js

JAVASCRIPT index.js 
require('dotenv').config();
const express = require('express');
const axios   = require('axios');
const OpenAI  = require('openai');

const app    = express();
const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });
app.use(express.json());

// In-memory store — swap for Redis in production
const conversations = {};

// ── Webhook verification
app.get('/webhook', (req, res) => {
  const { 'hub.mode': mode, 'hub.verify_token': token, 'hub.challenge': challenge } = req.query;
  if (mode === 'subscribe' && token === process.env.VERIFY_TOKEN)
    return res.send(challenge);
  res.sendStatus(403);
});

// ── Incoming messages
app.post('/webhook', async (req, res) => {
  res.sendStatus(200); // ACK immediately

  const msg = req.body?.entry?.[0]?.changes?.[0]?.value?.messages?.[0];
  if (!msg || msg.type !== 'text') return;

  const userId = msg.from;
  const text   = msg.text.body;

  const reply = await getAIReply(userId, text);
  await sendMessage(userId, reply);
});

// ── GPT-4 with conversation memory
async function getAIReply(userId, userMessage) {
  if (!conversations[userId]) conversations[userId] = [];

  const history  = conversations[userId].slice(-10); // keep last 10
  const messages = [
    { role: 'system', content: process.env.SYSTEM_PROMPT },
    ...history,
    { role: 'user', content: userMessage }
  ];

  const res = await openai.chat.completions.create({
    model:      process.env.OPENAI_MODEL,
    messages,
    max_tokens: 400
  });

  const reply = res.choices[0].message.content;
  conversations[userId].push({ role: 'user', content: userMessage });
  conversations[userId].push({ role: 'assistant', content: reply });
  return reply;
}

// ── Send text message
async function sendMessage(to, text) {
  await axios.post(
    `https://graph.facebook.com/v19.0/${process.env.WHATSAPP_PHONE_NUMBER_ID}/messages`,
    { messaging_product: 'whatsapp', to, type: 'text', text: { body: text } },
    { headers: { Authorization: `Bearer ${process.env.WHATSAPP_TOKEN}` } }
  );
}

app.listen(3000, () => console.log('Listening on :3000'));

02 Python Implementation

BASH terminal 
pip install flask openai requests python-dotenv
PYTHON app.py 
import os
from flask  import Flask, request, jsonify
from openai import OpenAI
import requests

app    = Flask(__name__)
client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])
conversations: dict[str, list] = {}

@app.route("/webhook", methods=["GET"])
def verify():
    mode      = request.args.get("hub.mode")
    token     = request.args.get("hub.verify_token")
    challenge = request.args.get("hub.challenge")
    if mode == "subscribe" and token == os.environ["VERIFY_TOKEN"]:
        return challenge, 200
    return "Forbidden", 403

@app.route("/webhook", methods=["POST"])
def webhook():
    data  = request.get_json()
    entry = data.get("entry", [{}])[0]
    msgs  = entry.get("changes", [{}])[0].get("value", {}).get("messages")
    if not msgs or msgs[0].get("type") != "text":
        return jsonify(ok=True)

    user_id = msgs[0]["from"]
    text    = msgs[0]["text"]["body"]
    reply   = get_ai_reply(user_id, text)
    send_message(user_id, reply)
    return jsonify(ok=True)

def get_ai_reply(user_id: str, user_msg: str) -> str:
    if user_id not in conversations:
        conversations[user_id] = []
    history  = conversations[user_id][-10:]
    messages = [
        {"role": "system",    "content": os.environ["SYSTEM_PROMPT"]},
        *history,
        {"role": "user", "content": user_msg}
    ]
    res   = client.chat.completions.create(model=os.environ["OPENAI_MODEL"], messages=messages, max_tokens=400)
    reply = res.choices[0].message.content
    conversations[user_id] += [{"role":"user","content":user_msg},{"role":"assistant","content":reply}]
    return reply

def send_message(to: str, text: str):
    pid   = os.environ["WHATSAPP_PHONE_NUMBER_ID"]
    token = os.environ["WHATSAPP_TOKEN"]
    requests.post(
        f"https://graph.facebook.com/v19.0/{pid}/messages",
        headers={"Authorization": f"Bearer {token}"},
        json={"messaging_product":"whatsapp","to":to,"type":"text","text":{"body":text}}
    )

if __name__ == "__main__":
    app.run(port=3000)

03 Adding Interactive Reply Buttons

Plain text replies work, but tap-able buttons significantly improve UX and conversion.

JAVASCRIPT buttons.js 
async function sendButtons(to, bodyText, buttons) {
  await axios.post(
    `https://graph.facebook.com/v19.0/${process.env.WHATSAPP_PHONE_NUMBER_ID}/messages`,
    {
      messaging_product: 'whatsapp', to,
      type: 'interactive',
      interactive: {
        type: 'button',
        body: { text: bodyText },
        action: {
          buttons: buttons.map((label, i) => ({
            type:  'reply',
            reply: { id: `btn_${i}`, title: label }
          }))
        }
      }
    },
    { headers: { Authorization: `Bearer ${process.env.WHATSAPP_TOKEN}` } }
  );
}

// Example usage
await sendButtons(userId, "How can I help you today?", [
  "Check order status",
  "Product info",
  "Talk to agent"
]);
Note

WhatsApp supports max 3 buttons per message (Reply Buttons) or up to 10 options via List Messages. For a detailed comparison of when each type converts better, see the business guide linked below.

04 Human Handover Pattern

Every production bot needs a reliable escalation path. Detect trigger words and route to a human agent before the conversation breaks down.

JAVASCRIPT handover.js 
const TRIGGERS = [
  'human', 'agent', 'manager', 'complaint', 'refund',
  'not helpful', 'speak to someone',
  // Arabic
  'موظف', 'مدير', 'استرداد', 'شكوى'
];

function needsHandover(text) {
  return TRIGGERS.some(t => text.toLowerCase().includes(t));
}

// In your main handler:
if (needsHandover(text)) {
  await sendMessage(userId,
    "I'll connect you with a team member right away. 🙏\n" +
    "Someone will be with you shortly."
  );
  await notifyHumanAgent(userId, text); // your internal alert
  return;
}

05 Production Checklist

  • Replace conversations {} with Redis (TTL: 24h)
  • Verify webhook signature (X-Hub-Signature-256)
  • Rate limit per userId (prevent abuse)
  • Add fallback reply for OpenAI timeout
  • Log conversations to DB for analytics
  • Handle media messages (images, voice notes)
  • Monitor Quality Rating in WhatsApp Manager
Cost Note

Replies sent within 24h of a customer message are free on Meta's side. Only messages you initiate (broadcasts, notifications) incur fees — ~0.17 SAR/marketing message for Saudi numbers. OpenAI costs ~$0.01–0.02 per conversation with gpt-4o-mini.
