01 What This Dashboard Tracks

WhatsApp Business API gives you raw webhook events. This project aggregates them into metrics that actually inform campaign decisions.

Why these four matter

Delivery and read rates confirm infrastructure health. Response rate is the conversion metric. Opt-out rate is the quality signal — anything above 2% on a single campaign requires immediate review.

02 Architecture Overview

// Backend — Node.js / Express

JAVASCRIPTserver/index.js
const express  = require('express');
const mongoose = require('mongoose');
const app = express();
app.use(require('cors')());
app.use(express.json());

// ── Receive WhatsApp status events
app.post('/webhook', async (req, res) => {
  const statuses =
    req.body?.entry?.[0]?.changes?.[0]?.value?.statuses;

  if (statuses) {
    for (const s of statuses) {
      await MessageEvent.findOneAndUpdate(
        { messageId: s.id },
        { status: s.status, timestamp: s.timestamp,
          recipientId: s.recipient_id },
        { upsert: true }
      );
    }
  }
  res.sendStatus(200);
});

mongoose.connect(process.env.MONGO_URI)
  .then(() => app.listen(4000));

03 React Frontend — Metrics Hook

JAVASCRIPTsrc/hooks/useCampaignStats.js
import { useState, useEffect } from 'react';

export function useCampaignStats(campaignId) {
  const [stats,   setStats  ] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const load = async () => {
      const res  = await fetch(`/api/campaigns/${campaignId}/stats`);
      const data = await res.json();
      setStats(data); setLoading(false);
    };
    load();
    const t = setInterval(load, 30_000); // poll every 30s
    return () => clearInterval(t);
  }, [campaignId]);

  return { stats, loading };
}

// MongoDB Aggregation Pipeline

JAVASCRIPTserver/routes/campaigns.js
const stats = await MessageEvent.aggregate([
  { $match: { campaignId: req.params.id } },
  {
    $group: {
      _id:       null,
      total:     { $sum: 1 },
      delivered: { $sum: { $cond: [{ $eq: ['$status','delivered']},1,0] }},
      read:      { $sum: { $cond: [{ $eq: ['$status','read']},1,0] }},
      replied:   { $sum: { $cond: ['$replied',1,0] }},
      optedOut:  { $sum: { $cond: ['$optedOut',1,0] }}
    }
  },
  {
    $project: {
      deliveryRate: { $divide: ['$delivered','$total'] },
      readRate:     { $divide: ['$read','$delivered'] },
      responseRate: { $divide: ['$replied','$read'] },
      optOutRate:   { $divide: ['$optedOut','$total'] }
    }
  }
]);
